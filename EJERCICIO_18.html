# Required libraries for ARIMA modeling and data analysis
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.tsa.stattools import adfuller
from statsmodels.graphics.tsaplots import plot_acf, plot_pacf
from statsmodels.tsa.arima.model import ARIMA
import warnings
from google.colab import files  # Required for file upload in Google Colab
warnings.filterwarnings("ignore")

# Step 1: Upload the Excel File
# Explanation: Use files.upload() to prompt the user to upload an Excel file with a button.
print("Please upload an Excel file containing the macroeconomic data.")
uploaded = files.upload()
filename = list(uploaded.keys())[0]
data = pd.read_excel(filename)

# Step 2: Select the Column for Analysis
# Explanation: Ask the user to specify which column (e.g., 'PDI') they want to analyze.
print("Available columns in your dataset:", data.columns.tolist())
column_name = input("Enter the name of the column you want to analyze (e.g., 'PDI'): ")

# Step 3: Transform the Data to Log Form
# Explanation: Apply the natural logarithm to the selected column to stabilize variance,
# as the problem specifies using log DPI.
data['Log_' + column_name] = np.log(data[column_name])

# Step 4: Visualize the Time Series
# Explanation: Plot the log-transformed data to inspect trends and seasonality visually.
plt.figure(figsize=(10, 6))
plt.plot(data['Log_' + column_name], label='Log ' + column_name)
plt.title('Log-Transformed ' + column_name + ' Over Time')
plt.xlabel('Time')
plt.ylabel('Log ' + column_name)
plt.legend()
plt.show()

# Step 5: Check Stationarity with ADF Test
# Explanation: Perform the Augmented Dickey-Fuller test to check if the series is stationary.
# If not, differencing may be required.
def adf_test(series):
    result = adfuller(series.dropna())
    print('ADF Statistic:', result[0])
    print('p-value:', result[1])
    print('Critical Values:', result[4])
    if result[1] <= 0.05:
        print("The series is stationary (reject the null hypothesis).")
    else:
        print("The series is non-stationary (fail to reject the null hypothesis).")

print("\nADF Test on Log " + column_name + ":")
adf_test(data['Log_' + column_name])

# Step 6: Difference the Series if Necessary
# Explanation: If the series is non-stationary, take the first difference and recheck stationarity.
data['Diff_Log_' + column_name] = data['Log_' + column_name].diff()
print("\nADF Test on Differenced Log " + column_name + ":")
adf_test(data['Diff_Log_' + column_name])

# Visualize the differenced series
plt.figure(figsize=(10, 6))
plt.plot(data['Diff_Log_' + column_name], label='Differenced Log ' + column_name)
plt.title('Differenced Log ' + column_name + ' Over Time')
plt.xlabel('Time')
plt.ylabel('Differenced Log ' + column_name)
plt.legend()
plt.show()

# Step 7: Identify ARIMA Parameters (p, d, q) Using ACF and PACF
# Explanation: Plot the Autocorrelation Function (ACF) and Partial Autocorrelation Function (PACF)
# to determine the AR (p) and MA (q) terms. The differencing order (d) is determined from stationarity.
plt.figure(figsize=(12, 4))
plt.subplot(121)
plot_acf(data['Diff_Log_' + column_name].dropna(), ax=plt.gca(), lags=20)
plt.title('ACF of Differenced Log ' + column_name)
plt.subplot(122)
plot_pacf(data['Diff_Log_' + column_name].dropna(), ax=plt.gca(), lags=20)
plt.title('PACF of Differenced Log ' + column_name)
plt.show()

# Step 8: Fit the ARIMA Model
# Explanation: Fit an ARIMA model based on the identified parameters.
# For demonstration, we assume ARIMA(1,1,1) as a starting point; adjust p, d, q based on ACF/PACF.
p, d, q = 1, 1, 1  # These values should be adjusted based on ACF/PACF analysis
model = ARIMA(data['Log_' + column_name], order=(p, d, q))
results = model.fit()

# Print the model summary
print("\nARIMA Model Summary:")
print(results.summary())

# Step 9: Diagnose the Model Residuals
# Explanation: Check the residuals to ensure they resemble white noise (no autocorrelation).
residuals = results.resid
plt.figure(figsize=(10, 6))
plt.plot(residuals, label='Residuals')
plt.title('Residuals of ARIMA Model')
plt.xlabel('Time')
plt.ylabel('Residuals')
plt.legend()
plt.show()

# ACF of residuals
plt.figure(figsize=(10, 4))
plot_acf(residuals, lags=20)
plt.title('ACF of Residuals')
plt.show()

# Step 10: Forecast Future Values (Optional)
# Explanation: Generate forecasts for the next few periods to evaluate the models predictive power.
forecast = results.forecast(steps=5)
print("\nForecasted Log " + column_name + " for the next 5 periods:")
print(forecast)

# Plot the forecast
plt.figure(figsize=(10, 6))
plt.plot(data['Log_' + column_name], label='Observed Log ' + column_name)
plt.plot(forecast, label='Forecasted Log ' + column_name, color='red')
plt.title('Log ' + column_name + ' Forecast')
plt.xlabel('Time')
plt.ylabel('Log ' + column_name)
plt.legend()
plt.show()

# Final Note
# Explanation: Summarize the process and suggest refining the model if needed.
print("\nSteps completed! Review the ACF/PACF plots and residuals to refine p, d, q if necessary.")
